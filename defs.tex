% % Custom quote environment
\renewenvironment{quote}[1]{%
  \def\quoteauthor{#1}% store author name
  \begin{list}{}{%
    \leftmargin0.3cm
    \rightmargin\leftmargin
  }%
  \item\relax
  \itshape ``%
}{% end-code
  ’’% closing quote
  \hspace{0.2cm}\normalfont\small\hfill{--- \quoteauthor}%
  \end{list}%
}

%%% Use this for quotes where authors name goes to the second line
\newenvironment{quotel}[1]{%
  \def\quoteauthor{#1}% store author name
  \begin{list}{}{%
    \leftmargin0.3cm
    \rightmargin\leftmargin
  }%
  \item\relax
  \itshape``% opening quote (no space after)
}{% end-code
  ''% closing quote (no space before)
  \par\vspace{-0.5ex}% move author to new line, tighten spacing
  \normalfont\small\hfill--- \quoteauthor%
  \end{list}%
}
